Standard Binary Search

				
Binary Search finds the index of a target element in a sorted array. If the element is found, it returns the index; otherwise, it returns -1.

				
                

                    

                        
                        
                    

                    

                        
                        
1.	Set low to 0 and high to the last index of the array.

2.	While low is less than or equal to high:
	o	Calculate the middle index mid.
	o	If array[mid] equals the target, return mid.
	o	If array[mid] is less than the target, search in the right half (low = mid + 1).
	o	If array[mid] is greater than the target, search in the left half (high = mid - 1).

						


                       
                    

                    
                    
                

				
				
				
The time complexity of standard binary search is: O(log n)

Here's why:
1.Binary search works by dividing the search space in half after each comparison.
2.Starting with a list of n elements, each step reduces the number of elements to search by half.
3.This halving continues until the element is found or the search space is empty.

Since the number of divisions required to reduce n elements to 1 element is approximately O(log n), 
the time complexity of binary search is O(log n).

#include 
using namespace std;

int binarySearch(int arr[], int size, int target) {
    int low = 0, high = size - 1;
    while (low <= high) {
        int mid = (low + high) / 2;
        if (arr[mid] == target) return mid;
        else if (arr[mid] < target) low = mid + 1;
        else high = mid - 1;
    }
    return -1;
}

int main() {
    int arr[] = {1, 3, 5, 7, 9};
    int target = 5;
    int size = sizeof(arr) / sizeof(arr[0]);
    cout << "Index of target: " << binarySearch(arr, size, target) << endl;  // Output: 2
    return 0;
}

                        


                        
public class BinarySearch {
    public static int binarySearch(int[] arr, int target) {
        int low = 0, high = arr.length - 1;
        while (low <= high) {
            int mid = (low + high) / 2;
            if (arr[mid] == target) return mid;
            else if (arr[mid] < target) low = mid + 1;
            else high = mid - 1;
        }
        return -1;
    }

    public static void main(String[] args) {
        int[] arr = {1, 3, 5, 7, 9};
        int target = 5;
        System.out.println("Index of target: " + binarySearch(arr, target));  // Output: 2
    }
}

                        


                        
def binary_search(arr, target):
    low, high = 0, len(arr) - 1
    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            low = mid + 1
        else:
            high = mid - 1
    return -1

# Example
arr = [1, 3, 5, 7, 9]
target = 5
print("Index of target:", binary_search(arr, target))  # Output: 2
