Linear Search

				
Linear Search is a simple searching algorithm that checks each element in an array or list one by one until it finds the target element or reaches the end of the list..

				
                

                    

                        
                        
                    

                    

                        
                        
1.	Start from the first element of the array.
2.	Compare each element with the target value.
3.	If you find the target, return its index.
4.	If you reach the end of the array without finding the target, return -1.


						


                       
                    

                    
                    
                

				
				
				
Time Complexity

•	Worst Case: O(n), where nnn is the number of elements in the array. This happens if the target is at the end of the list or is not in the list at all.
•	Best Case: O(1), if the target is the first element.
•	Average Case: O(n).

Space Complexity
•	O(1) because linear search only requires a constant amount of extra space for variables.

#include 
using namespace std;

int linearSearch(int arr[], int size, int target) {
    for (int i = 0; i < size; i++) {
        if (arr[i] == target) {
            return i;  // Return the index of the target
        }
    }
    return -1;  // If the target is not found
}

int main() {
    int arr[] = {10, 23, 45, 70, 11, 15};
    int target = 70;
    int size = sizeof(arr) / sizeof(arr[0]);
    cout << "Index of target: " << linearSearch(arr, size, target) << endl;  // Output: 3
    return 0;
}


                        


                        
public class LinearSearch {
    public static int linearSearch(int[] arr, int target) {
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == target) {
                return i;  // Return the index of the target
            }
        }
        return -1;  // If the target is not found
    }

    public static void main(String[] args) {
        int[] arr = {10, 23, 45, 70, 11, 15};
        int target = 70;
        System.out.println("Index of target: " + linearSearch(arr, target));  // Output: 3
    }
}


                        


                        
def linear_search(arr, target):
    for index, element in enumerate(arr):
        if element == target:
            return index  # Return the index of the target
    return -1  # If the target is not found

# Example usage
arr = [10, 23, 45, 70, 11, 15]
target = 70
print("Index of target:", linear_search(arr, target))  # Output: 3
