Double LinkedList Searcing

				
A Doubly Linked List (DLL) is a type of linked list where each node contains three components:


1. Data

2. A pointer to the previous node (prev)

3. A pointer to the next node (next).

				
                

                    

                        
                        
                    

                    

                        
                        
1. Initilize current to point to the head of the list.
2. While current is not NULL:
	If current->data equals the target element, return the node or its position.
	Move current to current->next.
3. If the end of the list is reached without finding the element, return NULL or indicate the element is not found.

#include 
using namespace std;

// Define a node for the doubly linked list
struct Node {
    int data;
    Node* prev;
    Node* next;
};

// Function to search for an element in the doubly linked list
Node* search(Node* head, int target) {
    Node* current = head;
    while (current != NULL) {
        if (current->data == target) {
            return current; // Element found
        }
        current = current->next;
    }
    return NULL; // Element not found
}

int main() {
    // Example usage
    Node* head = new Node{10, NULL, NULL};
    Node* second = new Node{20, head, NULL};
    Node* third = new Node{30, second, NULL};
    head->next = second;
    second->next = third;

    int target = 20;
    Node* result = search(head, target);
    if (result) {
        cout << "Element " << target << " found.\n";
    } else {
        cout << "Element " << target << " not found.\n";
    }
    return 0;
}


                        


                        
class DoublyLinkedList {
    static class Node {
        int data;
        Node prev, next;

        Node(int data) {
            this.data = data;
            this.prev = null;
            this.next = null;
        }
    }

    public static Node search(Node head, int target) {
        Node current = head;
        while (current != null) {
            if (current.data == target) {
                return current; // Element found
            }
            current = current.next;
        }
        return null; // Element not found
    }

    public static void main(String[] args) {
        Node head = new Node(10);
        Node second = new Node(20);
        Node third = new Node(30);

        head.next = second;
        second.prev = head;
        second.next = third;
        third.prev = second;

        int target = 20;
        Node result = search(head, target);
        if (result != null) {
            System.out.println("Element " + target + " found.");
        } else {
            System.out.println("Element " + target + " not found.");
        }
    }
}
